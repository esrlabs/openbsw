39  bsp::BspReturnCode init(tFtmCombinePwmConfiguration const* const config = nullptr)
40  {
41  if (config != nullptr)
42  {
43  _configuration = config;
44  }
45  _ftm.writeProtectionDisable(true);
46  _HwChannel0.sc = 0;
47  _HwChannel1.sc = 0;
48 
49  _HwChannel0.sc = _HwChannel0.sc | ((_configuration->ctrl0 & tFtm::CTRL_ELSx_MASK) << 2);
50  _HwChannel0.sc = _HwChannel0.sc | ((_configuration->ctrl0 & tFtm::CTRL_MSx_MASK) << 2);
51  _HwChannel1.sc = _HwChannel1.sc | ((_configuration->ctrl1 & tFtm::CTRL_ELSx_MASK) << 2);
52  _HwChannel1.sc = _HwChannel1.sc | ((_configuration->ctrl1 & tFtm::CTRL_MSx_MASK) << 2);
53  clrEvent();
54 
55  // combine link settings
56 
57  uint8_t comb = 0;
58  if ((_configuration->ctrl0 & tFtm::CTRL_COMBINE_MASK) != 0)
59  {
60  comb |= tFtm::COMBINE_COMB;
61  }
62  if ((_configuration->ctrl0 & tFtm::CTRL_MCOMBINE_MASK) != 0)
63  {
64  comb |= tFtm::COMBINE_MCOMBINE;
65  }
66  if ((_configuration->ctrl0 & tFtm::CTRL_DECAPEN_MASK) != 0)
67  {
68  comb |= tFtm::COMBINE_DECAPEN;
69  }
70  comb |= tFtm::COMBINE_SYNCEN;
71  _ftm.setCombine((hwChannel / 2) * 2, comb);
72 
73  _ftm.setCPWMS((_configuration->ctrl0 & tFtm::CTRL_CPWMS_MASK) != 0);
74 
75  (void)Io::setDefaultConfiguration(_configuration->pin);
76 
77  return bsp::BSP_OK;
78  }

153  inline uint32_t getDuty()
154  {
155  uint32_t const per = static_cast<uint32_t>(_ftm.getMOD() - _ftm.getCNTIN() + 1);
156  uint16_t const c0 = static_cast<uint16_t>(_HwChannel0.v);
157  uint16_t const c1 = static_cast<uint16_t>(_HwChannel1.v);
158  uint32_t duty = 0;
159  if (c1 >= c0)
160  {
161  duty = static_cast<uint32_t>(c1 - c0);
162  }
163  if (c0 < _ftm.getCNTIN())
164  {
165  duty = 0;
166  }
167  if (duty >= per)
168  {
169  return static_cast<uint32_t>(100 * res);
170  }
171  else
172  {
173  return duty * 100 * res / per;
174  }
175  }

177  inline void setDuty(uint16_t duty)
178  {
179  uint16_t const range = res * 100;
180  if (duty > range)
181  {
182  duty = range;
183  }
184  uint16_t const mod = _ftm.getMOD();
185  uint16_t const cint = _ftm.getCNTIN();
186  if (duty == 0)
187  {
188  _HwChannel0.v = static_cast<uint32_t>(mod);
189  _HwChannel1.v = static_cast<uint32_t>(mod);
190  }
191  else if (duty == range)
192  {
193  _HwChannel1.v = static_cast<uint32_t>(mod + 1);
194  _HwChannel0.v = static_cast<uint32_t>(cint);
195  }
196  else
197  {
198  uint16_t const per = mod - cint + 1;
199  uint16_t const lduty = static_cast<uint16_t>(static_cast<uint32_t>(per) * duty / range);
200  if (lduty < 0xFFFFU)
201  {
202  _HwChannel0.v = static_cast<uint32_t>(cint + (mod - cint) / 2 - lduty / 2);
203  _HwChannel1.v = static_cast<uint32_t>(cint + (mod - cint) / 2 + lduty / 2);
204  }
205  }
206  forceReload();
207  }
