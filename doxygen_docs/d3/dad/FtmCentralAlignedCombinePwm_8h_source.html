73  else

76  uint16_t const lduty = static_cast<uint16_t>(static_cast<uint32_t>(per) * duty / range);
77  if (lduty < 0xFFFFU)
78  {
79  _ch.setC0(cint + (mod - cint) / 2 - lduty / 2);
80  _ch.setC1(cint + (mod - cint) / 2 + lduty / 2);
