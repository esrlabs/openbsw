125  {

139  inline void setDuty(uint16_t duty)
140  {
141  uint16_t const range = res * 100;
142  if (duty > range)
143  {
144  duty = range;
145  }
146  if (duty == 0)
147  {
148  _HwChannel.v = _ftm.getCNTIN();
149  }
150  else if (duty == range)
151  {
152  _HwChannel.v = static_cast<uint32_t>(_ftm.getMOD() + 1);
153  }
154  else
155  {
156  uint16_t const per = _ftm.getMOD() - _ftm.getCNTIN() + 1;
157  uint16_t const lduty = static_cast<uint16_t>(
158  static_cast<uint32_t>(per) * duty / range + _ftm.getCNTIN());
159  if (lduty < 0xFFFFU)
160  {
161  _HwChannel.v = lduty;
162  }
163  }
