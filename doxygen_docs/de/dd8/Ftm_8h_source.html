181  inline void
182  setCombine(uint8_t channel, bool const combine, bool const mcombine, bool const decapen)
183  {
184  if (channel < maxChannels)
185  {
186  if ((channel & 0x01U) == 0)
187  {
188  channel >>= 1;
189  channel *= 8;
190  uint32_t comb = 0;
191  if (combine)
192  {
193  comb |= static_cast<uint32_t>(COMBINE_COMB);
194  }
195  if (mcombine)
196  {
197  comb |= static_cast<uint32_t>(COMBINE_MCOMBINE);
198  }
199  if (decapen)
200  {
201  comb |= static_cast<uint32_t>(COMBINE_DECAPEN);
202  }
203  uint32_t mask = static_cast<uint32_t>(0xFFU) << channel;
204  uint32_t ist_combine = _ftm.COMBINE;
205  ist_combine &= ~mask;
206  uint32_t sollV = comb << channel;
207  ist_combine |= sollV;
208  _ftm.COMBINE = ist_combine;
209  }
210  }
211  }
