950 template<class T>
951 inline void forward_list<T>::remove(forward_list_node<T>& node)
952 {
953  if (empty())
954  {
955  return;
956  }
957 
958  auto current = before_begin();
959  auto previous = end();
960  auto const endIter = end();
961  while ((current != endIter) && (current._current != &node))
962  {
963  previous = current;
964  ++current;
965  }
966 
967  if (current == endIter)

995 template<class T>

997 void forward_list<T>::remove_if(Predicate pred, Removed& removed)
998 {
999  forward_list tmp;
1000  iterator inItr = tmp.before_begin();
1001  while (!empty())
1002  {
1003  reference current = front();
1004  pop_front();
1005  if (pred(current) == false)
1006  {
1007  (void)tmp.insert_after(inItr, current);
1008  ++inItr;
1009  }
1010  else
1011  {
1012  removed.push_front(current);
1013  }
1014  }
1015  forward_list::swap(tmp);
1016 }

1096 template<class T>
1097 void forward_list<T>::splice_after(
1098  iterator const position, iterator const first, iterator const last)
1099 {
1100  if (position != first)
1101  {
1102  iterator element = ::estd::next(first);
1103  if ((position != element) && (last != element))
1104  {
1105  // first points to last after this
1106  first._current->set_next(last._current);
1107 
1108  node_* const tmp = position._current->next();
1109  position._current->set_next(element._current);
1110  while (element._current->next() != last._current)
1111  {
1112  ++element;
1113  }
1114  element._current->set_next(tmp);
1115  }
1116  }

1137 {

1139

1141  {
1142  return endIter;
1143  }
1144

1146  node_* const next = tmp->next();
1147  if (next == nullptr)

1222 template<class Compare>
1223 void forward_list<T>::sort(Compare const compare)
1224 {
1225  if (empty())
1226  { // nothing to sort in an empty list
1227  return;
1228  }
1229  auto list = begin();
1230  auto const endIter = end();
1231 
1232  size_type list_size = 1U;
1233  size_type num_merges = 0U;
1234 
1235  do
1236  {
1237  num_merges = 0U;
1238  iterator left = list;
1239  iterator tail = endIter;
1240  list = endIter;
1241  iterator next;
1242  while (left != endIter)
1243  {
1244  ++num_merges;
1245  iterator right = left;
1246  size_t left_size = 0U;
1247  size_t right_size = list_size;
1248  // cut lists into halves
1249  while ((right != endIter) && (left_size < list_size))
1250  {
1251  ++left_size;
1252  ++right;
1253  }
1254  // run through lists and append
1255  while ((left_size > 0U) || ((right_size > 0U) && (right != endIter)))
1256  {
1257  if (left_size == 0U)
1258  { // left is empty -> take right
1259  next = right;
1260  ++right;
1261  --right_size;
1262  }
1263  else if ((right_size == 0U) || (right == endIter))
1264  { // right is empty -> take left
1265  next = left;
1266  ++left;
1267  --left_size;
1268  }
1269  else if (compare(*right, *left) == true)
1270  { // right goes before left
1271  next = right;
1272  ++right;
1273  --right_size;
1274  }
1275  else
1276  { // left goes before right
1277  next = left;
1278  ++left;
1279  --left_size;
1280  }
1281  if (tail != endIter)
1282  {
1283  tail->node_::set_next(next._current);
1284  }
1285  else
1286  {
1287  list = next;
1288  }
1289  tail = next;
1290  }
1291  // right is now after the just sorted part, start next round here
1292  left = right;
1293  }
1294  // terminate tail
1295  tail->node_::set_next(nullptr);
1296  // double list size
1297  list_size <<= 1U;
1298  } while (num_merges > 1U);
1299  _before_first.set_next(list._current);
1300 }
