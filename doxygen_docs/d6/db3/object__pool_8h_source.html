596 template<class T>
597 inline typename object_pool<T>::value_type*
598 object_pool<T>::next(value_type const* const current) const
599 {
600  if (current == nullptr)
601  {
602  return nullptr;
603  }
604  if (!contains(*current))
605  {
606  return nullptr;
607  }
608  size_type const offset = static_cast<size_type>(current - type_utils<T>::cast_to_type(_data));
609  for (size_type i = offset + 1; i < max_size(); ++i)
610  {
611  if (isUsed(i))
612  {
613  return type_utils<T>::cast_to_type(_data, i);
614  }
615  }
616  return nullptr;
617 }
