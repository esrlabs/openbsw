961 template<class T>
962 inline void
963 vector<T>::insert(const_iterator const position, size_type const n, const_reference value)
964 {
965  estd_assert((size() + n) <= max_size());
966  iterator insertItr = begin() + (position - cbegin());
967  auto const diffToEnd = static_cast<difference_type>(cend() - position);
968  auto const numToMove = static_cast<size_type>(diffToEnd);
969 
970  if (n > numToMove) // appended elements will consist of moved ones and new copies of 'value'
971  {
972  // append new copies of 'value'
973  for (size_t i = 0; i < n - numToMove; ++i)
974  {
975  (void)emplace_back().construct(value);
976  }
977  // append new elements that are copies of existing ones
978  iterator srcItr = insertItr;
979  for (size_t i = n - numToMove; i < n; ++i)
980  {
981  (void)emplace_back().construct(::std::move(*srcItr));
982  ++srcItr;
983  }
984  // create remaining new copies of 'value' at the given position
985  for (size_t i = 0; i < numToMove; ++i)
986  {
987  *insertItr = value;
988  ++insertItr;
989  }
990  }
991  else // appended elements will consist of moved ones only
992  {
993  // append new elements that are copies of existing elements
994  iterator srcItr = end() - n;
995  for (size_t i = 0; i < n; ++i)
996  {
997  (void)emplace_back().construct(::std::move(*srcItr));
998  ++srcItr;
999  }
1000  // make more room by moving existing elements in reverse order
1001  srcItr = end() - 2U * n - 1;
1002  iterator dstItr = end() - n - 1;
1003  for (size_t i = 0; i < numToMove - n; ++i)
1004  {
1005  *dstItr = ::std::move(*srcItr);
1006  --dstItr;
1007  --srcItr;
1008  }
1009  // create new copies of 'value' at the given position
1010  for (size_t i = 0; i < n; ++i)
1011  {
1012  *insertItr = value;
1013  ++insertItr;
1014  }
1015  }
